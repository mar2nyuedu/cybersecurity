CREATE PROCEDURE InsertUser(IN In_ROW_NUMBER INT, IN In_ID VARCHAR(10), IN In_PASSWORD VARCHAR(10))
MODIFIES SQL DATA
BEGIN ATOMIC
    INSERT INTO "USER" ("ROW_NUMBER", "ID", "PASSWORD") VALUES (In_ROW_NUMBER, In_ID, In_PASSWORD);
END;

CREATE PROCEDURE FindUser(IN i_ID VARCHAR(10), IN i_PASSWORD VARCHAR(10))
MODIFIES SQL DATA
BEGIN ATOMIC
    -- Declare a cursor variable to hold the result set

 DECLARE r SCROLL CURSOR WITHOUT HOLD WITH RETURN FOR 
        SELECT * FROM "USER" WHERE "ID" = i_ID AND "PASSWORD" = i_PASSWORD;

    -- Initialize the cursor variable
    OPEN r;
END;
